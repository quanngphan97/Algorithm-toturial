#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int N;//Number of sponsored goods
int D[100000 + 10];//Preference
int sum = 0;

void InputData(){
	cin >> N;
	for(int i = 0; i < N; i++) {
		cin >> D[i];	
	}
}

void Solve()
{	
	int findMaxHere = 0;
	
	int maxIndex = INT_MIN;
	
	for(int i = 0 ; i < N; i++) {
		
		findMaxHere += D[i];
		
		findMaxHere = max(findMaxHere,0);
		
		sum = max(sum,findMaxHere);
		
		if(maxIndex < D[i]) 
				maxIndex = D[i];
	
	}
	
	if(sum == 0) {
		
		cout << maxIndex << endl;
	
	} else {
		
		cout << sum << endl;
	
	}
	
}


int main(){
	InputData();//Input function
	Solve();
	return 0;
}
